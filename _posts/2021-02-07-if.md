## if
```js
if(condition){
  code;
}
```
## if - else
```js
if(condition){
  code;
} else{
  code;
}
```
## if - else if
```js
if(conditionA){
  code;
} else if(conditionB){
  code;
} else{
  code;
}
```
## switch/case
```js
switch(value){
  case conditionA:
    code;
    break;
  case conditoinB:
    code;
    break;
  default:
    code;
}
```
